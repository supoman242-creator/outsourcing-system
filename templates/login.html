// login.html 내의 로그인 함수 예시 async function handleLogin(e) { e.preventDefault(); const id = document.querySelector('input[name="username"]').value; const pw = document.querySelector('input[name="password"]').value; const res = await fetch('/api/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ id: id, pw: pw }) // app.py의 id, pw와 맞춤 }); if (res.ok) { location.href = '/'; } else { alert('로그인 실패!'); } }
